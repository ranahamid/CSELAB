#include <iostream>
#include <math.h>
#include <iomanip>
 
using namespace std;
 
const double pi = 2*acos(0);
 
int main ()
{
    int D, V;
    double d;
    while (cin >> D >> V)
    {
        if (!(D + V)) return 0;
        d = pow ((D*D*D*pi-6*V)/pi, 1./3.);
        cout << fixed << setprecision(3) << d << endl;
    }
    return 0;
}
