Department: Information TechnologySemester 4
College: K.J. Somaiya College of Engineering

Hamming.java
  1. import java.util.*;
  2. import java.lang.*;
  3. class Hamming
  4. {
  5. 	public static void main(String args[])
  6. 	{
  7. 		int i=0,j,d=0,sum=0;
  8. 		int a[]=new int[7];
  9. 		int b[]=new int[11];
  10. 		int c[]=new int[11];
  11. 		int p[]=new int[4];
  12. 		Scanner sr=new Scanner(System.in);
  13. 		System.out.println("Enter the data code:");
  14. 		for(i=0;i<7;i++)
  15. 		{
  16. 			a[i]=sr.nextInt();
  17. 		}
  18. 		System.out.println("");
  19. 		System.out.println("data code is:");
  20.  
  21. 		for(i=0;i<7;i++)
  22.  
  23. 		{
  24. 			System.out.print(a[i]);
  25. 		}
  26.  
  27. 		for(j=0;j<11;j++)
  28. 		{
  29. 			b[j]=0;
  30.  
  31. 		}
  32.  
  33. 		if(((a[0]+a[1]+a[3]+a[4]+a[6])%2)==0)
  34. 		b[0]=0;
  35. 		else
  36. 		b[0]=1;
  37.  
  38. 		if(((a[0]+a[2]+a[3]+a[5]+a[6])%2)==0)
  39. 		b[1]=0;
  40. 		else
  41. 		b[1]=1;
  42.  
  43. 		if(((a[1]+a[2]+a[3])%2)==0)
  44. 		b[3]=0;
  45. 		else
  46. 		b[3]=1;
  47.  
  48. 		if(((a[4]+a[5]+a[6])%2)==0)
  49. 		b[7]=0;
  50. 		else
  51. 		b[7]=1;
  52.  
  53. 		System.out.println("");	
  54. 		for(j=0,i=0;j<11;)	
  55. 		{
  56. 			if(j==0||j==1||j==3||j==7)
  57. 			j++;
  58. 			else
  59. 			{
  60. 				b[j]=a[i];
  61. 				j++;
  62. 				i++;
  63. 			}			
  64. 		}
  65. 		System.out.println("");
  66. 		System.out.println("Hamming code for even parity is:");  
  67. 		for(j=0;j<11;j++)
  68. 		{
  69.  
  70. 			System.out.print(b[j]);
  71. 		}
  72.  
  73. 		System.out.println("");
  74. 		System.out.println("");
  75. 		System.out.println("Enter hamming code: ");
  76.  
  77. 		for(i=0;i<11;i++)
  78. 		{
  79. 			c[i]=sr.nextInt();
  80. 		}
  81.  
  82. 		p[0]=c[0];
  83. 		p[1]=c[1];
  84. 		p[2]=c[3];
  85. 		p[3]=c[7];
  86.  
  87. 		if((((c[2]+c[4]+c[6]+c[8]+c[10])%2)==0 && p[0]==0)||(((c[2]+c[4]+c[6]+c[8]+c[10])%2)!=0 && p[0]==1))
  88. 		p[0]=0;
  89. 		else
  90. 		p[0]=1;
  91.  
  92. 		if((((c[2]+c[5]+c[6]+c[9]+c[10])%2)==0 && p[1]==0)||(((c[2]+c[5]+c[6]+c[9]+c[10])%2)!=0 && p[1]==1))
  93. 		p[1]=0;
  94. 		else
  95. 		p[1]=1;
  96.  
  97. 		if((((c[4]+c[5]+c[6])%2)==0 && p[2]==0)||(((c[4]+c[5]+c[6])%2)!=0 && p[2]==1))
  98. 		p[2]=0;
  99. 		else
  100. 		p[2]=1;
  101.  
  102. 		if((((c[8]+c[9]+c[10])%2)==0 && p[3]==0)||(((c[8]+c[9]+c[10])%2)!=0 && p[3]==1))
  103. 		p[3]=0;
  104. 		else
  105.  
  106. 		p[3]=1;			
  107. 		for(i=3;i>=0;i--)
  108. 		{			
  109. 			d=d+(p[i]*(int)Math.pow(2,i));			
  110. 		}
  111. 		System.out.println("");		
  112. 		if(d==0)
  113. 		System.out.println("The data code is correctly received");	
  114. 		else
  115. 		{
  116. 			System.out.println("The "+d+" bit is wrongly received");
  117. 			if(c[d-1]==0)
  118. 			c[d-1]=1;
  119. 			else
  120. 			c[d-1]=0;
  121. 			System.out.println("");
  122. 			System.out.println("The correct hamming code is ");
  123. 			for(i=0;i<11;i++)
  124. 			{
  125. 				System.out.print(c[i]);
  126. 			}
  127. 		}
  128. 	}		
  129. }
